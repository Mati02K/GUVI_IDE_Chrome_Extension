document.addEventListener("paste", (event) => {
    let paste = event.clipboardData.getData('text');
    var lookalike = isLikeWallet(paste)
    if (lookalike) {
        chrome.storage.sync.get("wallets", (items) => {
            var wallets = items.wallets;
            if(!wallets.includes(paste)) {
                alert('alert')
            }
        });
    }
});
